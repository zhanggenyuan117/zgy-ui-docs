</template>

<script lang="ts" setup>
import { Select } from "zgy-ui";
import { ref } from "vue";

const selectvalue2 = ref("2")

  {
    label:"天爱",
    value:1
  },
  {
    label:"娜扎",
    value:2
  },
])

</script>
